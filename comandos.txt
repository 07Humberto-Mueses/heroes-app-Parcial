  export const getGifs = async (query) => {
    
    const url = `https://api.unsplash.com/search/photos?query=${query}&orientation=landscape&client_id=${accessKey}`;
    const resp = await fetch(url);
    const { results } = await resp.json();

    const gifs = results.map(photo => ({
        id: photo.id,
        title: photo.alt_description || 'No title',
        url: photo.urls.regular,
        user: {
            name: photo.user.name,
            profile: `https://unsplash.com/@${photo.user.username}`
        }
    }));

    return gifs;
};
